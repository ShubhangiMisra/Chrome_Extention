function fetch_feed("https://api.twitter.com/*", callback) { var xhr = new XMLHttpRequest(); xhr.onreadystatechange = function(data) { if (xhr.readyState == 4) { if (xhr.status == 200) { var data = xhr.responseText; callback(data); } else { callback(null); } } } // Note that any URL fetched here must be matched by a permission in // the manifest.json file! xhr.open('GET', "https://api.twitter.com/*", true); xhr.send(); } function onRequest(request, sender, callback) { if (request.action == 'fetch_feed') { fetch_feed(request."https://api.twitter.com/*", callback); } } // Wire up the listener. chrome.extension.onRequest.addListener(onRequest);
